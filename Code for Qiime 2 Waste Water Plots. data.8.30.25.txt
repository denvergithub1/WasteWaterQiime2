# Now for taxonomic analysis. For this, qiime2 uses what
## is called a Naive Bayes classifier which finds the 
### operational taxonomic unit (OTU). So from the V4 region, we are 
#### going to include only 250 bp from the trimmed region

wget \
  -O 'gg-13-8-99-515-806-nb-classifier.qza' \
  'https://docs.qiime2.org/2021.11/data/tutorials/moving-pictures-usage/gg-13-8-99-515-806-nb-classifier.qza'

qiime feature-classifier classify-sklearn \
  --i-classifier gg-13-8-99-515-806-nb-classifier.qza \
  --i-reads rep-seqs.qza \
  --o-classification taxonomy.qza
qiime metadata tabulate \
  --m-input-file taxonomy.qza \
  --o-visualization taxonomy.qzv

# With this visual, we can go into the blast hits we got

  --m-metadata-file sample-metadata.tsv \
  --o-visualization taxa-bar-plots.qzv
  

#################################
#Finally in the qiime2 package, we can consider doing the 
##differential abundance testing with ANCOM (analysis of 
### composition of microbes)It compares the composition
#### microbes in all our populations here. 
##### This method gives improved inference in microbiome data, even better 
###### than t.tests which can give inflated comparison significance. 

qiime feature-table filter-samples \
  --i-table table.qza \
  --m-metadata-file sample-metadata.tsv \
  --p-where '[reverse-index]='"'"'TTCTAGCT'"'"'' \
  --o-filtered-table AllSamples-table.qza

qiime composition add-pseudocount \
  --i-table AllSamples-table.qza \
  --o-composition-table comp-AllSamples-table.qza



# Now using the artifacts created above, we can use ANCOM 
## to determine what features are different in abundance
### across sample0 (or any other sample we put in place)

qiime composition ancom \
  --i-table comp-AllSamples-table.qza \
  --m-metadata-file sample-metadata.tsv \
  --m-metadata-column forward-index \ 
  --o-visualization ancom-AllSamples.qzv
